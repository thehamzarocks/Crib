# Blocks
3.times {puts "Hello"}

2.times do |index| # index acts as a parameter here
	if index > 0
		puts index
	end
end

# or 

2.times {|index| puts index if index > 0 } # one-liner


# Blocks in methods

def twice_implicit
	return "no block" unless block_given? # implicit checking whether a block was provided
	yield # "calls" the block
	yield
end

puts twice_implicit {print "Hello"} # Hello printed twice in seperate lines
puts twice_implicit # no block

# or

def twice_explicit(&i_block) # use & before the last parameter
	return "no block" if i_block.nil?
	i_block.call
	i_block.call
end
